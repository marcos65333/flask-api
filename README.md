# flask-api
 how create a api with Flask using python

# CREATE  INSTALL VIRTUAL VENV 
  * python3 -m venv venv

# INSTALL FLASK
  * pip install flask

# RUN FLASK
  * python app.py

# TYPES REQUESTS "CRUD"
  * GET -> READ
  * POST --> CREATE
  * PUT -> UPDATE
  * DELETE -> DELETE

# ENVIROMENT VARIABLES
  * CREATE .env
  * PIP INSTALL python-dotenv

# DATABASE  REQUIREMENTS
  * pip install mysql-connector
